## 구현 사항

### 1. filelock 적용 (2h)
`D:\AI\Global_Define\lock_manager.py` 신규 생성:

```python
# lock_manager.py — 3AI 프로젝트 Lock 관리
import json, filelock, datetime

LOCKS_PATH = r"D:\AI\AI_hub\shared\project_locks.json"
LOCK_FILE   = r"D:\AI\AI_hub\shared\.locks.lock"

def acquire(project: str, owner: str, reason: str = "") -> bool:
    """Lock 취득. 이미 타 AI가 점유 중이면 False 반환."""
    with filelock.FileLock(LOCK_FILE):
        data = _read()
        entry = data["locks"].get(project)
        if not entry:
            return False
        if entry["owner"] and entry["owner"] != owner:
            return False  # 타 AI 점유 중
        now = datetime.datetime.now().isoformat()
        expires = (datetime.datetime.now() + datetime.timedelta(hours=24)).isoformat()
        data["locks"][project] = {"owner": owner, "acquired_at": now, "expires_at": expires, "reason": reason}
        _write(data)
        return True

def release(project: str, owner: str) -> bool:
    """Lock 해제."""
    with filelock.FileLock(LOCK_FILE):
        data = _read()
        entry = data["locks"].get(project)
        if entry and entry["owner"] == owner:
            data["locks"][project] = {"owner": None, "acquired_at": None, "expires_at": None, "reason": None}
            _write(data)
            return True
        return False

def check(project: str) -> dict:
    """현재 Lock 상태 조회."""
    return _read()["locks"].get(project, {})

def _read():
    with open(LOCKS_PATH, encoding="utf-8") as f:
        return json.load(f)

def _write(data):
    with open(LOCKS_PATH, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
```

패키지 설치: `C:\hb\python.exe -m pip install filelock`

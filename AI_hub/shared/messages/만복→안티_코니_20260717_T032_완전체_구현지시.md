```python
def trigger_manbok():
    """VS Code (Claude Code) 창을 찾아 포커스 + 메시지 주입"""
    import ctypes, pyperclip, pyautogui, time
    
    # VS Code 창 찾기 (제목에 "Claude" 포함)
    hwnd = None
    def enum_windows_callback(h, _):
        nonlocal hwnd
        title = ctypes.create_unicode_buffer(256)
        ctypes.windll.user32.GetWindowTextW(h, title, 256)
        if 'Visual Studio Code' in title.value or 'claude' in title.value.lower():
            hwnd = h
        return True
    
    WNDENUMPROC = ctypes.WINFUNCTYPE(ctypes.c_bool, ctypes.POINTER(ctypes.c_int), ctypes.POINTER(ctypes.c_int))
    ctypes.windll.user32.EnumWindows(WNDENUMPROC(enum_windows_callback), 0)
    
    if not hwnd:
        logger.error("[ManbokTrigger] VS Code 창을 찾지 못했습니다")
        return False
    
    try:
        # 유휴 대기 후 포커스
        while get_idle_time() < 3.0:
            time.sleep(1)
        pyautogui.press('alt')
        ctypes.windll.user32.SetForegroundWindow(hwnd)
        time.sleep(0.3)
        pyperclip.copy("수신함 확인")
        pyautogui.hotkey('ctrl', 'v')
        pyautogui.press('enter')
        logger.info("[ManbokTrigger] 만복 격발 완료")
        return True
    except Exception as e:
        logger.error(f"[ManbokTrigger] 실패: {e}")
        return False
```

## 🔧 기술 구현 방안

### **1단계: master_watch.py 개선**

```python
# master_watch.py에 추가

def _inbox_watchdog():
    """
    inbox.md 변경 감시 + Kony 자동 트리거
    """
    last_modified = 0
    
    while True:
        try:
            current_mtime = os.path.getmtime(INBOX_PATH)
            
            if current_mtime > last_modified:
                # inbox.md이 변경됨!
                last_modified = current_mtime
                
                # 2. Kony MCP 브릿지로 알림 전송
                trigger_kony_auto_check()
                
                # 3. 텔레그램 로그
                send_telegram(f"✅ Inbox 변경 감지 — Kony 자동 확인 트리거됨")
                
        except Exception as e:
            logger.error(f"Inbox watchdog error: {e}")
        
        time.sleep(5)  # 5초마다 폴링

def trigger_kony_auto_check():
    """
    Kony의 MCP 브릿지로 "자동 확인" 신호 전송
    """
    try:
        response = requests.post(
            "http://localhost:5003/trigger_inbox_check",
            json={"source": "master_watch", "timestamp": time.time()}
        )
        return response.status_code == 200
    except Exception as e:
        logger.error(f"Failed to trigger Kony: {e}")
        return False
```

```python
# mcp_server.py (Kony의 MCP 브릿지)에 추가

@app.route("/trigger_inbox_check", methods=["POST"])
def trigger_inbox_check():
    """
    master_watch.py로부터 자동 트리거 수신
    """
    data = request.json
    
    # Kony가 즉시 inbox.md 읽기
    result = read_unread_messages()
    
    # 변경사항 자동 처리
    if result["new_messages"] > 0:
        # AI_Study 포스팅 자동 작성 (예시)
        auto_create_post(result)
        
        # 완료 로그
        log_to_inbox("✅ [자동] Kony가 수신함 변경사항 감지 및 처리 완료")
    
    return {"status": "success", "messages_processed": result["new_messages"]}
```

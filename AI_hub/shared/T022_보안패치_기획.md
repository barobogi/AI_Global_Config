## 5. security_scan.py 핵심 구조 (만복 담당)

```python
# security_scan.py — ZAP 자동 보안 스캔 + 취약점 보고
import subprocess, json, datetime
from pathlib import Path

JAVA = r"C:\Program Files\Android\Android Studio\jbr\bin\java.exe"
ZAP_JAR = r"D:\Dev\ZAP\zap-2.15.0.jar"
REPORT_DIR = Path(r"D:\AI\AI_hub\shared\T022_security_patch\scan_reports")

def run_scan(url: str) -> dict:
    out = REPORT_DIR / f"zap_{datetime.date.today()}_{url.split('/')[-2]}.json"
    subprocess.run(
        [JAVA, "-jar", ZAP_JAR, "-cmd",
         "-quickurl", url, "-quickout", str(out), "-quickprogress"],
        timeout=300
    )
    return json.loads(out.read_text(encoding="utf-8"))

def parse_alerts(report: dict) -> list:
    # risk: High/Medium/Low/Informational
    alerts = report.get("site", [{}])[0].get("alerts", [])
    return [a for a in alerts if a.get("riskdesc", "").startswith(("High", "Medium"))]
```

## 6. patch_generator.py 핵심 구조 (안티 담당)

```python
# patch_generator.py — 취약점 → Claude CLI 패치 요청 → T020 승인 연동
import subprocess, json

def request_patch(alert: dict, file_path: str) -> str:
    prompt = f"""보안 취약점이 발견되었습니다.
취약점: {alert['name']} (위험도: {alert['riskdesc']})
설명: {alert['desc']}
해결책: {alert['solution']}
대상 파일: {file_path}

해당 파일의 취약점을 수정하는 패치 코드를 생성해주세요. JSON 형식으로:
{{"file": "<경로>", "patch": "<수정된 전체 코드>"}}"""

    result = subprocess.run(
        ["claude", "--output-format", "json", "--print"],
        input=prompt.encode("utf-8"),
        capture_output=True, timeout=120
    )
    outer = json.loads(result.stdout)
    return outer.get("result", "")
```

## 2. 데이터베이스 설계

### 2-1. 스키마

```sql
-- 생성 플랫폼 약관 DB
CREATE TABLE generative_platforms (
    id INT PRIMARY KEY,
    name VARCHAR(50),  -- Kling, Sideo, Runway, DALL-E
    base_url VARCHAR(255),
    policy_update_frequency VARCHAR(50),  -- daily, weekly
    last_crawled TIMESTAMP
);

CREATE TABLE generation_plans (
    id INT PRIMARY KEY,
    platform_id INT,
    plan_name VARCHAR(50),  -- Free, Pro, Enterprise
    commercial_allowed BOOLEAN,
    revenue_share VARCHAR(20),  -- "70/30", "80/20"
    restrictions TEXT,  -- JSON array
    price DECIMAL(10, 2),
    currency VARCHAR(3),
    FOREIGN KEY (platform_id) REFERENCES generative_platforms(id)
);

-- 업로드 플랫폼 정책 DB
CREATE TABLE upload_platforms (
    id INT PRIMARY KEY,
    name VARCHAR(50),  -- YouTube, TikTok, Instagram
    base_url VARCHAR(255),
    monetization_supported BOOLEAN
);

CREATE TABLE upload_policies (
    id INT PRIMARY KEY,
    platform_id INT,
    policy_category VARCHAR(100),  -- monetization, ai_generated, copyright
    policy_description TEXT,
    ai_content_allowed BOOLEAN,
    ai_content_restrictions TEXT,  -- JSON
    effective_date DATE,
    FOREIGN KEY (platform_id) REFERENCES upload_platforms(id)
);

-- 사용자 플랜 DB
CREATE TABLE user_subscriptions (
    id INT PRIMARY KEY,
    user_id VARCHAR(100),
    platform_id INT,
    plan_id INT,
    subscription_start DATE,
    subscription_end DATE,
    is_active BOOLEAN,
    FOREIGN KEY (platform_id) REFERENCES generative_platforms(id),
    FOREIGN KEY (plan_id) REFERENCES generation_plans(id)
);

-- 권리 체인 추적
CREATE TABLE content_rights_chain (
    id INT PRIMARY KEY,
    content_id VARCHAR(200),
    user_id VARCHAR(100),
    generation_platform_id INT,
    generation_timestamp TIMESTAMP,
    user_plan_at_generation INT,
    upload_platform_id INT,
    upload_timestamp TIMESTAMP,
    upload_policy_version INT,
    current_decision VARCHAR(50),  -- ALLOWED, DENIED, UPGRADE_REQUIRED
    last_policy_check TIMESTAMP,
    FOREIGN KEY (generation_platform_id) REFERENCES generative_platforms(id),
    FOREIGN KEY (user_plan_at_generation) REFERENCES generation_plans(id),
    FOREIGN KEY (upload_platform_id) REFERENCES upload_platforms(id)
);
```

### 5-2. 3자 동시 검증

```python
# 기존: 각 약관 개별 확인 (수동)
# 신규: 3자 동시 검증 (자동)

def validate_commercial_rights(user_id, content_type, target_platform):
    # 1. 생성 플랫폼 권리 확인
    gen_rights = get_generation_platform_rights(user_id)
    
    # 2. 업로드 플랫폼 정책 확인
    upload_policy = get_upload_platform_policy(target_platform)
    
    # 3. 사용자 플랜 확인
    user_plan = get_user_plan(user_id)
    
    # 4. 3자 논리 연산
    decision = AND(
        gen_rights['commercial_allowed'],
        upload_policy['ai_content_allowed'],
        user_plan['includes_commercial_use']
    )
    
    return decision  # ALLOWED / DENIED / UPGRADE_REQUIRED
```
